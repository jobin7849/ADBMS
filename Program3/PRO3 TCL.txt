1.COMMIT

mysql> set autocommit=0;
Query OK, 0 rows affected (0.05 sec)

mysql> insert into customer(age,name)values(26,'jobin');
Query OK, 1 row affected (0.05 sec)

mysql> commit;
Query OK, 0 rows affected (0.04 sec)

2.ROLLBACK

mysql> set autocommit=0;
Query OK, 0 rows affected (0.00 sec)

mysql>  insert into customer values(22,'nihad');
Query OK, 1 row affected (0.00 sec)

mysql> select * from customer;
+------+-------+
| age  | name  |
+------+-------+
|   26 | jobin |
|   22 | nihad |
+------+-------+
2 rows in set (0.00 sec)

mysql> rollback;
Query OK, 0 rows affected (0.04 sec)

mysql> select * from customer;
+------+-------+
| age  | name  |
+------+-------+
|   26 | jobin |
+------+-------+
1 row in set (0.00 sec)

3.SAVEPOINT

mysql> insert into customer values(23,'sreehari');
Query OK, 1 row affected (0.00 sec)

mysql> insert into customer values(22,'abhin');
Query OK, 1 row affected (0.00 sec)

mysql> update customer set age=25 where name='sreehari';
Query OK, 1 row affected (0.00 sec)
Rows matched: 1  Changed: 1  Warnings: 0

mysql> savepoint sp1;
Query OK, 0 rows affected (0.00 sec)

mysql> insert into customer values(24,'nafla');
Query OK, 1 row affected (0.00 sec)

mysql> savepoint sp2;
Query OK, 0 rows affected (0.00 sec)

mysql> delete from customer where age=23;
Query OK, 0 rows affected (0.00 sec)

mysql> rollback to sp2;
Query OK, 0 rows affected (0.00 sec)

mysql> select * from customer;
+------+----------+
| age  | name     |
+------+----------+
|   26 | jobin    |
|   25 | sreehari |
|   22 | abhin    |
|   24 | nafla    |
+------+----------+
4 rows in set (0.00 sec)

mysql> rollback to sp1;
Query OK, 0 rows affected (0.00 sec)

mysql> select * from customer;
+------+----------+
| age  | name     |
+------+----------+
|   26 | jobin    |
|   25 | sreehari |
|   22 | abhin    |
+------+----------+
3 rows in set (0.00 sec)

mysql> commit;
